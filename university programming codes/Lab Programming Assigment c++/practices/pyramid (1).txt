#include <iostream>
#include <conio.h>
using namespace std;

void main()
{
	int row,c,n,temp;
	cout << "Enter number of row \n";
	cin >> n;
	temp = n;
	for (row = 1; row <= n; row++)
	{
		for (c = 1; c < temp; c++)
			cout << " ";
			temp--;
			for (c = 1; c <=2*row-1; c++)
				cout << "x";
				cout << "\n";
	}
	getch();
}
