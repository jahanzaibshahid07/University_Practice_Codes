#include <iostream>
#include <conio.h>
#include <fstream>


using namespace std;

void main()
{
	int id;
	char name[20];

	ofstream outfile("player.txt");

	cout << "Enter id ";
	cin >> id;
	cout << "Enter name ";
	cin >> name;

	outfile << id << endl;
	outfile << name << endl;

	outfile.close();

	ifstream infile("data.txt");

	infile >> id;
	infile >> name;

	cout << "\nplayer id " << id; 
	cout << "\nplayer name " << name;

	infile.close();
 
	getch();
}
