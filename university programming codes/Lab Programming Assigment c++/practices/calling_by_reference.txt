void swap(int *a, int *b)
{
 int temp;
 temp=*a; //*a=>value of a.
 *a=*b;
 *b=temp;

 printf("\nThe Address of 'b' is  %x ",b);
 printf("\nThe value of 'a' is  %d ",*a);
 printf("\nThe value of 'b' is  %d ",*b);
}
