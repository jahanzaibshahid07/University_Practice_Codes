#include <iostream>
#include <conio.h>
using namespace std;

void parameter(int arr[]) // function
{
	for (int a = 0; a < 5; a++)
	{ 
		cout << arr[a] << endl; // output
	}
}

void main()
{
	int arr[5]; // assigned

	for (int a = 0; a < 5; a++)
	{
		cout << "Enter any array ";
		cin >> arr[a]; // array as input
	}

	parameter(arr); // function calling

	getch();
}
