#include <iostream>
#include <conio.h>

using namespace std;

struct Node               // node definition
{
	int data;
	Node *next;
};

struct Node *head,*p;   // globle variable

void InsertNodeAtTheFirstPostion(int value)  //function
{
	Node *temp = new Node;
	temp->data = value;               // insert at the first position
	temp->next = head;  
	head = temp;       
}

void Search(int item)
{
	int flag = 0;
	p = head;
	while(p->next != NULL)
	{
		if(p->data == item)
		{
			 cout << "item found : " << item;
			 flag = 1;
			 break;
		}
		else 
		{
			p = p->next;
		}
	}

	if(flag == 0)
	{
		cout << "item not found ";
	}
}



void PrintNode()
{
	Node *temp1 = head;
	
	if(temp1 == NULL)
	{
		cout << "list is empty";
	}
	else
	{
		cout << " List : ";
		while(temp1 != NULL)
		{
			cout << " " << temp1->data;
			temp1 = temp1->next;
		}
		cout << endl;
	}
	
}

void main()
{
	head = NULL;
	int n,x,item;
	
	cout << "how many number ";
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		cout << "Enter the number ";
		cin >> x;
		InsertNodeAtTheFirstPostion(x);
		PrintNode();
	}
	cout << "Enter item for search : ";
	cin >> item;
	Search(item);

}
