// Every Program by default has 1 Thread for executing all logics in the program 
// In multi-thread we create multiple thread for multiple logics so the program's logics execute parallel
// or simantaniously so the program delay time is less and response is fast 
// Background thread is destroyed when application is exist
// Foreground thread is still run when the application is exist

namespace InterviewPractice
{
    class Program
    {
        static void Test1()
        {
            for (int i = 1; i <= 100; i++)
            {
                Console.WriteLine("Test1 : " + i);
            }
            Console.WriteLine("Test1 is existing ");
        }

        static void Test2()
        {
            for (int i = 1; i <= 100; i++)
            {
                Console.WriteLine("Test2 : " + i);
                
                if (i == 50)
                {
                    Console.WriteLine("Test 2 is sleep");
                    Thread.Sleep(5000);
                    Console.WriteLine("Test 2 is woke up");
                }
            }
            Console.WriteLine("Test2 is existing ");
        }

        static void Test3()
        {
            for (int i = 1; i <= 100; i++)
            {
                Console.WriteLine("Test3 : " + i);
            }
            Console.WriteLine("Test3 is existing ");
        }

        static void Main(string[] args)
        {
            Thread t1 = new Thread(Test1);
            Thread t2 = new Thread(Test2);
            Thread t3 = new Thread(Test3);
            
            t1.Start();
            t2.Start();
            t3.Start();
            
            Console.WriteLine("Main is existing ");

            Console.ReadKey();
        }
    }
}
