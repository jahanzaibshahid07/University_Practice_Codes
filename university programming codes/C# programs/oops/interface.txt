// interface only declaration but no implementation.class is the implementation of interface.
// you cannot create an instance of the interface and abstuct classes as well..

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace consoleui
{
    interface Icustomer           // interface method cannot have access modifers.
    {
        void print();            //interface cannot have fields.
    }

    interface i2
    {
        void i2method();
    }

    class customer : Icustomer, i2  //class can inherit for multiple interfaces but class cannot inherit for multiple classes.
    {
        public void print()
        {
            Console.WriteLine("this is interface implementation");
        }

        public void i2method()
        {
            Console.WriteLine("i2 method second interface");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            customer c1 = new customer();
            c1.print();
            c1.i2method();

            Console.ReadKey();
        }
    }
}
