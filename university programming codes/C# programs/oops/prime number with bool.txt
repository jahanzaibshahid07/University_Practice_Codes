using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace ConsoleApplicationui
{ 
    class prime
    {
        public bool isprime(int x)
        {
            if (x <= 1) return false;

            for (int i = 2; i <= x / i; i++)
                if (x % i == 0) return false;

            return true;
        }
    }
       

    class Program
    {
        static void Main(string[] args)
        {
            prime p = new prime();

            for (int i = 2; i < 10; i++)
                if (p.isprime(i)) Console.WriteLine(i + " prime ");

                else Console.WriteLine(i + "not prime");
                Console.ReadLine();
        }
    }
}
