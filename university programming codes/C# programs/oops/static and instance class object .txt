// static class only have static methods and fields 
// static class don't create instance and called by class name
// singleton class can create 1 object but static can't create object

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace ConsoleApplicationUl
{
    class circle
    {
        static float _pi; // static give one copy of pi to allocate memory of thousand object and works efficiently.
        int _r;

        static circle()  // static constractor use to initiaze static field and dont use access modifier and work efficient
        {
            Console.WriteLine("Static constractor ");
            circle._pi = 3.14f;
        }

        public circle(int radius)  // instance constrator use access modifier by default private.  
        {
            Console.WriteLine("Instance constractor ");
            _r = radius;
        }

        public float CalCirleArea() // method
        {
            return circle._pi * (this._r * this._r); // static constructor are called before instance constructor..static cons only call once 
        }

    }

    class Program
    {
        static void Main(string[] args)
        {
            circle c1 = new circle(2);
            float area1 = c1.CalCirleArea();
            Console.WriteLine("Area = " + area1);


            circle c2 = new circle(5);
            float area2 = c2.CalCirleArea();
            Console.WriteLine("Area = " + area2);

            circle c3 = new circle(8);
            float area3 = c3.CalCirleArea();
            Console.WriteLine("Area = " + area3);

          Console.ReadKey();
        }
    }
}
